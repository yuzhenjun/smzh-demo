jqueryui.com/datepicker

初始化日历：datetimepicker();显示日期和时间  datepicker显示日期   timepicker显示时间
	$('#example').datetimepicker({
		timeFormat:'hh:mm:ss',//时间格式化 ：hh:mm:ss显示时分秒 hh:mm显示时分 hh 显示小时
		dateFormat:'yy-mm-dd',//日期格式化
		showOn: "button",//"button" for trigger button, or "both" for either
		buttonImageOnly: true,
		buttonImage: "../images/date.png"//以上为显示日期图标
		onSelect:function(){
			//添加逻辑
		}
	});
	使用div可以将日历显示在div块中
	设置时间：
	$('#example').datetimepicker( 'setDate' , new Date()) ;

	获取时间：
	$('#example').datetimepicker('getDate')
